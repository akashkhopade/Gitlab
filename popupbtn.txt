protected void GridView_RowCommand(object sender, GridViewCommandEventArgs e)
{
    if (e.CommandName == "ViewDetails")
    {
        int rowIndex = Convert.ToInt32(e.CommandArgument);
        // Retrieve data from the selected row
        string id = GridView1.Rows[rowIndex].Cells[0].Text;
        string name = GridView1.Rows[rowIndex].Cells[1].Text;
        // Set the data in the modal
        lblDetails.Text = "ID: " + id + "<br />Name: " + name;
        // Show the modal
        ScriptManager.RegisterStartupScript(this, this.GetType(), "myModal", "$('#myModal').modal();", true);
    }
}
